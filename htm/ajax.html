			Ambow.post({
				url:G_URL.list,
				params:{
					a:'111',
					b:'222'
				},
				success:function(data){
					alert(data.total);
				},
				error:function(){
					//如果返回的不是json格式会error
					alert('error');
				}
			});
		
			Ambow.getJSON(G_URL.list,{a:'1111'},function(data){
				alert(Ambow.encode(data));
			});
		
			Ambow.ajax({
				url:G_URL.list,
				method:'post',
				params:{
					a:'111',
					b:'222'
				},
				success:function(data){
					alert(data.total);
				},
				error:function(){
					//如果返回的不是json格式会error
					alert('error');
				}
			});
